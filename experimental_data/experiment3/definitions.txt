INSERT ASSIGNMENT(FIELD ACCESS x = ANY) INTO ANY | EXISTS(INSERT MEMBER m INTO ANY | x REFERS TO m);
INSERT ARGUMENT INTO ARGUMENTS a AT i | PARENT(a) IS CALL c AND
    EXISTS(INSERT PARAMETER INTO PARAMETERS p AT i | PARENT(p) IS METHOD m AND c REFERS TO m);
INSERT METHOD m INTO ANY | EXISTS(INSERT MEMBER mb INTO ANY | m IS GETTER FOR mb);
INSERT METHOD m INTO ANY | EXISTS(INSERT MEMBER mb INTO ANY | m IS SETTER FOR mb);
{
    INSERT METHOD m INTO ANY;
    INSERT TYPE INTO m;
    INSERT PARAMETERS INTO m;
    INSERT BLOCK b INTO m;
    MOVE ANY FROM ANY x TO b;
    INSERT CALL c INTO x | c REFERS TO m
}
